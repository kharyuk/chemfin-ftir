### Upsampling data to be of length 2^d
def upsample_data_degree2(data, freq):
    newN = int(2**np.floor(1+np.log2(data.shape[0])))
    newFreq = np.linspace(freq[0], freq[-1], newN)
    Y = []
    for k in xrange(Nsamples):
        finterp = scipy.interpolate.interp1d(freq, data[:, k], 'cubic')
        Y.append( finterp(newFreq) )
    newT = np.array(Y).T
    return newFreq, newT

def extractLinkageMatrix(model):
    '''
    This is adoptation of the original implementation
    available at
    https://github.com/scikit-learn/scikit-learn/blob/
    70cf4a676caa2d2dad2e3f6e4478d64bcb0506f7/examples/
    cluster/plot_hierarchical_clustering_dendrogram.py
    '''
    children = model.children_
    distance = model.distance
    num_observations = 2+np.arange(children.shape[0])
    linkageMatrix = np.column_stack([
        children, distance, num_observations
    ])
    return linkageMatrix

def snsHCAplot(ac_model, X, column_colors=None):
    
    method = ac_model.linkage
    metric = ac_model.affinity
    R = ac_model.contrast(X.T)
    col_linkage = extractLinkageMatrix(ac_model.alg)
    
    R = pairwise_distances(R, Y=None, metric=metric)
    cm = sns.clustermap(
        R, pivot_kws=None, method=method,
        metric=metric, z_score=False, standard_scale=None,
        figsize=(10, 3), cbar_kws=None, row_cluster=False, col_cluster=True,
        row_linkage=None, col_linkage=col_linkage, row_colors=None, col_colors=column_colors,
        mask=np.ones(R.shape), cmap="mako"
    )
    cm.fig.delaxes(cm.fig.axes[-2])
    cm.fig.delaxes(cm.fig.axes[-1])
    
    cm.fig.axes[-1].yaxis.set_visible(False)
    
    cm.gs.set_height_ratios([0.12, 0.1, 0.05, 0.001])
    cm.gs.set_width_ratios([0.001, 0.001, 0.8])

    cm.gs.update()
    return cm
